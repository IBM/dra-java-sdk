  public String getId() {
    return id;
  }

  /**
   * Gets the location.
   *
   * The location information of the DR workspace.
   *
   * @return the location
   */
  public LocationDr getLocation() {
    return location;
  }

  /**
   * Gets the name.
   *
   * The name of the DR workspace.
   *
   * @return the name

  public String getName() {
    return name;

   *
   * @return the status

  public String getStatus() {

}
